                  default: "true"

                  enum: [ "true", "false" ]
                  default: "false"

                  required: true

                  type: array

                    type: string
                    format: binary
                detached_files[]:
                  required: false
                  description: Array de arquivos destacados, respectivamente correspondentes às assinaturas em `signature_files[]`.

                  items:

                      - type: string

                          algorithm:
                            type: string

                            type: string
                            description: Valor do hash em formato hexadecimal

                    em `signature_files[]`.
                  type: array
                  items:
                    type: string
                    format: binary
                verification_packages[]:
                  required: false
                  description: Array de pacotes de verificação exportados em JSON ou em XML (XAIP)
                  type: array
                  items:
                    type: string
                    format: binary
                return_artifacts:
                  required: false
                  description: Quando habilitado, serão retornados todos os artefatos utilizados durante a verificação do documento fornecido. Estes artefatos incluem certificados, CRLs, OCSPs, referência de tempo, entre outros. De posse destes artefatos, é possível reproduzir a verificação do documento, sendo útil para fins de testes. O pacote de verificação pode ser baixado tanto em JSON (estrutura em array com uma entrada por arquivo verificado) quanto em XML no formato XAIP (pacote ZIP com um XML por arquivo verificado). Cada elemento destes conjuntos, por sua vez, contém pares chave-valor com informações relevantes e artefatos utilizados na verificação.O pacote é disponibilizado para download apenas para relatórios gerados em formato 'HTML', sendo também anexado ao JSON do relatório caso este esteja no formato 'JSON'.
                  type: boolean
                extensions_profile:
                  required: false
                  description: Visibilidade das informações do assinante
                  type: string
                  enum: [ "default", "visible", "masked", "invisible" ]
                  default: "default"
                profile_verification:
                  required: false

                  type: boolean
                  enum: [ "true", "false" ]
                  default: "false"
                signature_extension:
                  required: false
                  description: Se deve o verificador deve subir o nível do perfil da assinatura. A operação de subida de nível é feita apenas se profileVerification for true.

                  enum: [ "T", "LT", "LTA" ]
                tsa_info:
                  required: false
                  $ref: '#/components/schemas/TsaInfo'
                time_reference_type:
                  required: false
                  description: Tipo de referência de tempo a ser utilizada na verificação. Quando "current", utiliza a data/hora atual. Quando "custom", utiliza a data/hora informada em `time_reference`.
                  type: string
                  enum: ["current", "custom"]
                  default: "current"
                time_reference:
                  required: false
                  description: Data e hora customizadas para referência de tempo na verificação, no formato ISO 8601 (ex. `2025-06-15T14:30:00`). Utilizado quando `time_reference_type` é "custom".
                  type: string
                  format: date-time
      responses:
        200:
          description: Retorna o relatório de verificação de assinaturas. Os arquivos em formato PDF e XML têm organização similar ao esquema JSON fornecido. Em caso de recibo com um relatório diferente do HTML, retorna-se um arquivo ZIP contendo o recibo e o relatório no formato desejado
          content:
            application/json:
              schema:
                oneOf:
                  - $ref: '#/components/schemas/report'
                  - $ref: '#/components/schemas/simpleReport'
            application/pdf: { }
            text/xml: { }
        500:
          description: Erro na verificação de assinaturas. Favor utilizar o e-mail `cgnpe@iti.gov.br` para reportar o erro com o arquivo responsável em anexo.
        504:
          description: Erro na verificação de assinaturas. A verificação excedeu o limite de tempo.
        507:
          description: Erro na verificação de assinaturas. A verificação excedeu o limite de espaço de memória.

components:
  schemas:
    report:
      type: object
      description: Relatório do Verificador de Conformidade. É feito um relatório para cada arquivo de assinatura.
      properties:
        reports:
          type: array
          description: Relatório do Verificador de Conformidade. É feito um relatório para cada arquivo de assinatura.
          items:
            type: object
            properties:
              date:
                type: object
                description: Data de verificação
                properties:
                  sourceOfDate:
                    type: string
                    description: Como foi obtido (Online / Offline)
                  verificationDate:
                    type: string
                    description: Data de verificação do documento no formato 'dd/MM/yyyy HH:mm:ss zzz' (padrão Java)
                required:
                  - sourceOfDate
                  - verificationDate
              generalStatus:
                type: string
                description: Status do documento (Aprovado / Reprovado / Indeterminado / Incompleto)
              number:
                type: integer
                description: Número do relatório atual
              pas:
                type: object
                description: Informações de políticas de assinatura utilizadas no documento
                properties:
                  pa:
                    type: object
                    description: Informações de uma política de assinatura (doc-icp-15)
                    properties:
                      valid:
                        type: string
                        description: Se a PA é válida
                      period:
                        type: string
                        description: Período de validade da política, no formato "de 'dd/MM/yyyy HH:mm:ss zzz' até 'dd/MM/yyyy HH:mm:ss zzz'" (padrão Java)
                      expired:
                        type: boolean
                        description: Se a PA está com validade expirada
                      online:
                        type: boolean
                        description: Se foi obtida online
                      validOnLpa:
                        type: string
                        description: Íntegra segundo a LPA (Lista de políticas de assinatura aprovadas)
                      oid:
                        type: string
                        description: Identificador da política (nome e OID (Object Identifier) único)
                      revoked:
                        type: boolean
                        description: Se a PA foi revogada
                      error:
                        type: string
                        description: Mensagem de erro
                    required:
                      - error
                      - expired
                      - oid
                      - online
                      - period
                      - revoked
                      - valid
                      - validLpa
              software:
                type: object
                description: Sobre o software usado para verificar a assinatura
                properties:
                  name:
                    type: string
                    description: Nome do software (Verificador de Conformidade)
                  version:
                    type: number
                    description: Versão do Verificador de Conformidade utilizada
                  sourceFile:
                    type: string
                    description: Nome do arquivo fonte utilizado (p7s, xml, pdf)
                  sourceFileHash:
                    type: string
                    description: Resumo criptográfico SHA256 em base hexadecimal do arquivo fonte utilizado
                required:
                  - name
                  - sourceFile
                  - version
                  - sourceFileHash
              lpas:
                type: object
                description: Informações de listas de políticas de assinatura utilizadas no documento
                properties:
                  lpa:
                    type: object
                    description: Informações de uma lista de políticas de assinatura aprovadas (doc-icp-15)
                    properties:
                      name:
                        type: string
                        description: Nome da LPA, incluindo versão
                      online:
                        type: boolean
                        description: Se a LPA foi obtida online
                      valid:
                        type: boolean
                        description: Se a LPA é válida
                      lpaErrorMessage:
                        type: string
                        description: Mensagem de erro caso a lpa não esteja válida
                      period:
                        type: string
                        description: Data final de expiração da LPA no formato 'dd/MM/yyyy HH:mm:ss z' (padrão Java), ou uma mensagem de erro no lugar
                      expired:
                        type: boolean
                        description: Se a LPA está expirada
                      version:
                        type: string
                        description: Versão da LPA
                    required:
                      - name
                      - online
                      - valid
                      - period
                      - expired
                      - version
              initialReport:
                type: object
                description: Relatório inicial de dados da assinatura e arquivo
                properties:
                  qtdAnchorsSign:
                    type: integer
                    description: Quantidade de assinaturas ancoradas no arquivo
                  qtdSignatures:
                    type: integer
                    description: Quantidade de assinaturas no arquivo
                  fileType:
                    type: string
                    description: Tipo de arquivo identificado (ASN1, XML ou PDF)
                  errorMessages:
                    type: object
                    description: Mensagens de erro
                    properties:
                      errorMessage:
                        type: string
                        description: Mensagem de erro (se houver)
              onlyVerifyAnchored:
                type: boolean
                description: Se só serão consideradas assinaturas ancoradas no processo de verificação
              extendedReport:
                type: boolean
                description: Se o relatório é estendido (fazendo a verificação de todos os atributos) ou não (checando apenas caminho de certificação, cifra assimétrica e hash do arquivo)
              conformity:
                type: array
                description: Referências de conformidade encontradas no arquivo (caso assinatura seja válida)
                required: false
                items:
                  type: object
                  properties:
                    reference:
                      type: string
                      description: Referência que a validade da assinatura pertence
                    link:
                      type: string
                      description: Link para a validade da referência
              signatures:
                type: object
                description: Informações das assinaturas presentes no documento
                properties:
                  signature:
                    $ref: '#/components/schemas/signature'
              notAnchoredSignatures:
                type: object
                description: Informações das assinaturas não ancoradas presentes no documento
                properties:
                  notAnchoredSignature:
                    $ref: '#/components/schemas/signature'
              unprocessedSignatures:
                type: object
                description: Informações das assinaturas não processadas presentes no documento
                properties:
                  unprocessedSignature:
                    oneOf:
                      - $ref: '#/components/schemas/unprocessedSignature'
                      - type: array
                        description: Assinaturas não processadas
                        items:
                          $ref: '#/components/schemas/unprocessedSignature'
            required:
              - date
              - generalStatus
              - number
              - signatures
              - software
      required:
        - report
    configReport:

      properties:
        config:
          properties:

              type: string
              description: Tipo de relatório de verificação

              type: boolean
              description: Se o verificador deve realizar buscas por atualizações incrementais em documentos PDF

              type: boolean
              description: Se o verificador deve mostrar o conteúdo do payload de assinaturas JSON
            emitReceipt:
              type: boolean
              description: Se deve ser emitido um recibo da verificação
            returnArtifacts:
              type: boolean
              description: Quando habilitado, serão retornados todos os artefatos utilizados durante a verificação do documento fornecido. Estes artefatos incluem certificados, CRLs, OCSPs, referência de tempo, entre outros. De posse destes artefatos, é possível reproduzir a verificação do documento, sendo útil para fins de testes.

              generalStatus:

                type: integer
                description: Número do relatório atual
              software:
                type: object
                description: Sobre o software usado para verificar a assinatura

                  version:

                    description: Versão do Verificador de Conformidade utilizada

                    type: integer

                    description: Quantidade de assinaturas no arquivo
                  fileType:
                    type: string
                    description: Tipo de arquivo identificado (ASN1, XML ou PDF)

                    properties:

                      integrity:

                        properties:
                          schema:
                            type: boolean
                            description: Se é válida a estrutura da assinatura
                          schemaMessage:
                            type: string
                            description: Caso haja erro de conformidade com a estrutura da assinatura, este campo deve informar o motivo
                          schemaPattern:
                            type: string
                            description: Indica qual o padrão que a estrutura da assinatura deve estar de acordo
                          messageDigest:
                            type: string
                            description: Valor do MessageDigest gerado (do resumo criptográfico) de N caracteres (de acordo com tamanho da assinatura requirido)
                          references:
                            type: object
                            description: Referências nas assinaturas XML/XAdES (vazio em outros tipos de assinatura)
                            properties:

                                type: object

                          - hash
                          - messageDigest
                          - references
                          - schema
                          - schemaPattern
                      signatureType:

                          Pode ser vazio
                      signingTime:
                        type: string
                        description: Datetime da assinatura, se presente, no formato DateTimeFormatter Long (Java)

                                    description: Informações de identificação do assinante

                                        description: Informações do assinante

                                          type: object
                                          properties:
                                            name:

                                              description: Descrição da informação (E-mail, por exemplo)
                                            value:
                                              type: string
                                              description: Conteúdo da informação
                                          required:
                                            - name
                                            - value
                                  certificatePolicies:
                                    type: object
                                    description: Informações do certificado do assinante
                                    properties:
                                      certificatePolicy:
                                        type: array
                                        description: Informações do certificado
                                        items:
                                          type: object
                                          properties:
                                            name:
                                              type: string
                                              description: Descrição da informação (tipo do certificado, por exemplo)
                                            value:
                                              type: string
                                              description: Conteúdo da informação

                                          properties:
                                            validSignature:

                                              description: Se o certificado é válido segundo a resposta OCSP
